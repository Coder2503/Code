n=3
w=zeros(n-1, n);u=zeros(n-1, n);
P(1)=1;P(2)=x;
for i=2:n
P(i+1)=((2*i-1)*x*P(i)-(i-1)*P(i-1))/i
r=real(roots(P(i+1)));
for j=1:i
u(i-1,j)=r(j,1);
w(i-1,j)=2/((1-r(j, 1)^2)*(horner(derivat(P(i+1)),r(j, 1))^2));
end
end
